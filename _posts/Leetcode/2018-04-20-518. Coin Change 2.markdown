					class Solution():
					    def change(self, amount, coins):
					            """
					            :type amount: int
					            :type coins: List[int]
					            :rtype: int
					            """
					            dp = [1]+[0] * amount
					            #拿每个现有面额金币去分别拼成所有面额数的总数
					            for i in coins:
					                for j in range(1, amount + 1):
					                   if j >= i:
					                       dp[j] += dp[j - i]
					            return dp[amount]
					
					s = Solution()
					count = s.change(7, [1,2,5])
					print(count)
